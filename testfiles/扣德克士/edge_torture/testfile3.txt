/* edge_torture 3: for optional parts + multi-ForStmt + break/continue */
int main() {
    int i;
    int j;
    int k;
    int s = 0;

    for (i = 0, j = 10; i < 3; i = i + 1, j = j - 1) {
        s = s + i * 10 + j;
    }

    k = 0;
    for (; k < 5; ) {
        k = k + 1;
        if (k % 2 == 0) {
            continue;
        }
        s = s + k;
    }

    for (i = 0; ; i = i + 1) {
        if (i == 3) {
            break;
        }
        s = s + 100;
    }

    for (;;) {
        s = s + 1;
        break;
    }

    printf("%d\n", s);
    return 0;
}
