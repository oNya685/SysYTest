int data_pool[40000];

int processor(int p1[], int p2[], int p3[]) {
    int i = 1;
    int checksum = 0;
    for (;i < 40000;) {
        int temp = p1[i-1] + p3[i-1];
        p2[i] = temp * i;
        p1[i] = p2[i-1] / (i + 1);
        p3[i] = p1[i] - temp;
        checksum = checksum + p3[i];
        i = i + 1;
    }
    return checksum;
}

int main() {
    printf("23373270\n");
    int i = 0;
    for (i = 0; i < 40000; i = i + 1) {
        data_pool[i] = i + 1;
    }

    int res = processor(data_pool, data_pool, data_pool);

    printf("%d\n", res);
    printf("%d\n", data_pool[0]);
    printf("%d\n", data_pool[1]);
    printf("%d\n", data_pool[100]);
    printf("%d\n", data_pool[1000]);
    printf("%d\n", data_pool[20000]);
    printf("%d\n", data_pool[30000]);
    printf("%d\n", data_pool[39998]);
    printf("%d\n", data_pool[39999]);
    return 0;
}
