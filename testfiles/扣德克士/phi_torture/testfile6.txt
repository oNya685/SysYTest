/* phi_torture 6: if-with-inner-loop vs else, inner loop has continue/break */
int main() {
    int i;
    int j;
    int x = 0;
    int acc = 0;

    for (i = 0; i < 4; i = i + 1) {
        int y = i;

        if (i % 2 == 0) {
            for (j = 0; j < 5; j = j + 1) {
                if (j == 1) {
                    y = y + 10;
                    continue;
                }
                if (j == 3) {
                    y = y + 100;
                    break;
                }
                y = y + j;
            }
        } else {
            j = 7;
            y = y + 1000;
        }

        x = x + y + j;
        acc = acc + x;
    }

    printf("%d\n", x * 100 + acc);
    return 0;
}
