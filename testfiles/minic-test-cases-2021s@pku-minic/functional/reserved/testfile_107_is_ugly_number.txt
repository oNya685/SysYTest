int main()         
{
 	int num=5400;
 	int flag;
    for(; num % 2 == 0;) {
        num = num / 2;
    }
    for(; num % 3 == 0;) {
        num = num / 3;
    }
    for(; num % 5 == 0;) {
        num = num / 5;
    }
    if (num == 1) {
        flag=1;
    } else {
        flag=0;
    }
    return flag;
}
