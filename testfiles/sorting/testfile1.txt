// 冒泡排序实现
int bubbleSort(int arr[], int n) {
    int i;
    int j;
    int temp;
    
    for (i = 0; i < n - 1; i = i + 1) {
        for (j = 0; j < n - 1 - i; j = j + 1) {
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
    
    return 0;
}

int main() {
    int n;
    int arr[100];
    int i;
    
    n = getint();
    
    for (i = 0; i < n; i = i + 1) {
        arr[i] = getint();
    }
    
    bubbleSort(arr, n);
    
    for (i = 0; i < n; i = i + 1) {
        printf("%d", arr[i]);
        if (i < n - 1) {
            printf(" ");
        }
    }
    printf("\n");
    
    return 0;
}
