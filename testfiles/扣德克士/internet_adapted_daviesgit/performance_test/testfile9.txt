// Variant of testfile7: exercises "continue" in the inner loop to create extra join points.

int x;

const int N = 14;
const int NN = N * N;
const int MOD = 10007;

int A[NN] = {};
int B[N] = {};
int C[N] = {};

int prng() {
    static int s = 2025;
    s = (s * 77 + 55) % MOD;
    return s;
}

void init_vec(int n, int v[], int salt) {
    int i = 0;
    for (i = 0; i < n; i = i + 1) {
        v[i] = (prng() + salt + i) % 31;
        if (v[i] == 0) v[i] = 2;
    }
}

void init_mat(int n, int a[], int salt) {
    int nn = n * n;
    int i = 0;
    for (i = 0; i < nn; i = i + 1) {
        int v = (prng() + salt + (i % 9)) % 23;
        if (i % (n + 1) == 0) v = 0;
        a[i] = v;
    }
}

void mv(int n, int A[], int b[], int res[]) {
    int i, j;
    int y = 0;

    for (i = 0; i < n; i = i + 1) res[i] = 0;

    {
        int x = 11;
        for (i = 0; i < n; i = i + 1) {
            for (j = 0; j < n; j = j + 1) {
                int v = A[i * n + j];
                if (v == 0) {
                    x = (x * b[i] + b[j]) % MOD;
                    y = y - x;
                    if (y < 0) y = y + MOD;
                    continue;
                }
                res[i] = (res[i] + v * b[j]) % MOD;
            }
            y = (y + res[i]) % MOD;
        }
        y = (y + x) % MOD;
    }

    x = (x + y) % MOD;
}

int main() {
    x = 0;
    init_mat(N, A, 4);
    init_vec(N, B, 8);

    int it = 0;
    for (it = 0; it < 12; it = it + 1) {
        mv(N, A, B, C);
        mv(N, A, C, B);
    }

    int ans = x;
    int i = 0;
    for (i = 0; i < N; i = i + 1) {
        ans = (ans + (B[i] % 257)) % 257;
    }
    printf("%d\n", ans);
    return 0;
}
