int a[12];

int main() {
    int i = 0;
    int cnt = 0;
    for (; i <= 3 + 4 - 2; ) {
        int j = i;
        for (; j >= 0; ) {
            if (j < 4 && i - j < 3) {
                a[(i - j) * 4 + j] = cnt;
                cnt = 1;
            }
            j = j - 1;
        }
        i = i + 1;
    }
    printf("%d\n", a[0] + a[11] + cnt);
    return 0;
}
