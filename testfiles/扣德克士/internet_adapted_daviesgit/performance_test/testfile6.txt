// Adapted from DaviesGit/SysY_compiler:
// https://raw.githubusercontent.com/DaviesGit/SysY_compiler/master/test_sample/performance_test/01_mm3.sy
// Variant of testfile4: uses nested blocks and slightly different update order.

const int N = 11;
const int NN = N * N;
const int MOD = 10007;

int A[NN] = {};
int B[NN] = {};
int C[NN] = {};

int prng() {
    static int s = 7;
    s = (s * 73 + 19) % MOD;
    return s;
}

void init_mat(int n, int a[], int salt) {
    int nn = n * n;
    int i = 0;
    for (i = 0; i < nn; i = i + 1) {
        int v = (prng() + salt + (i % 5)) % 19;
        if (i % (n + 3) == 0) v = 0;
        a[i] = v;
    }
}

void mm(int n, int A[], int B[], int C[]) {
    int i, j, k;

    for (i = 0; i < n; i = i + 1) {
        for (j = 0; j < n; j = j + 1) {
            C[i * n + j] = 0;
        }
    }

    for (k = 0; k < n; k = k + 1) {
        for (i = 0; i < n; i = i + 1) {
            int aik = A[i * n + k];
            if (aik == 0) continue;

            {
                int base = i * n;
                for (j = 0; j < n; j = j + 1) {
                    int idx = base + j;
                    int tmp = C[idx] + aik * B[k * n + j];
                    C[idx] = tmp % MOD;
                }
            }
        }
    }
}

int main() {
    init_mat(N, A, 2);
    init_mat(N, B, 9);

    int it = 0;
    for (it = 0; it < 3; it = it + 1) {
        mm(N, A, B, C);
        mm(N, A, C, B);
        mm(N, B, A, C);
    }

    int ans = 0;
    int i = 0;
    for (i = 0; i < NN; i = i + 1) {
        ans = (ans + (C[i] % 997)) % 997;
    }
    printf("%d\n", ans);
    return 0;
}
