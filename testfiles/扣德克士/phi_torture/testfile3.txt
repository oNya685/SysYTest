/* phi_torture 3: inner-loop exit phi (break vs normal), using j/local after loop */
const int lim[3] = {2, 10, 1};

int main() {
    int i;
    int x = 0;

    for (i = 0; i < 3; i = i + 1) {
        int j;
        int local = 0;

        for (j = 0; j < 5; j = j + 1) {
            local = local + 1;
            if (j == lim[i]) {
                x = x + 100 + j;
                break;
            }
            x = x + 10;
        }

        x = x + j + local;
    }

    printf("%d\n", x);
    return 0;
}
