int a[25] = {1, 2, 3, 4, 5};

int func(int a[]) {
    int i = 0;
    int j = 0;
    int sum = 0;
    for (; i < 5; ) {
        for (; j < 5; ) {
            sum = sum + a[i * 5 + j];
            j = j + 1;
        }
        i = i + 1;
    }
    return sum;
}

int main() {
    printf("%d\n", func(a));
    return 0;
}
