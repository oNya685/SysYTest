/* phi_torture 2: continue edge into loop header (multiple preds for phi) */
int main() {
    int i;
    int x = 0;
    int y = 0;

    for (i = 0; i < 8; i = i + 1) {
        if (i % 3 == 0) {
            x = x + 100;
            continue;
        }

        if (i % 2 == 0) {
            x = x + 1;
        } else {
            x = x + 2;
        }
        y = y + x;
    }

    printf("%d\n", x * 10 + y);
    return 0;
}
