int lengthOfLastWord(int s[], int n) {
    if(n == 0)
        return 0;
    int c;
    c = n - 1;
    for (; c > -1 && s[c] == 0; ){
        c = c - 1;
    }
    if(c == -1)
        return 0;
    int i;
    i = c; 
    for (; i > -1; ){
        if(s[i] == 0)
            return n - i - 1 - (n - 1 - c);
        i = i - 1;
    }
    return c - i;
}
