/* phi_torture 8: nested loops + inner break/continue + outer continue */
int main() {
    int i;
    int j;
    int x = 0;
    int y = 0;

    for (i = 0; i < 4; i = i + 1) {
        int t = 0;

        for (j = 0; j < 6; j = j + 1) {
            if (j == i) {
                t = t + 100;
                continue;
            }

            if (t > 110) {
                t = t + 1000;
                break;
            } else {
                t = t + j;
            }

            if (j % 2 == 0) {
                x = x + t;
            } else {
                x = x - t;
            }
        }

        if (t % 2 == 0) {
            y = y + t + j;
            continue;
        } else {
            y = y - t + j;
        }

        x = x + y;
    }

    printf("%d\n", x * 10 + y);
    return 0;
}
