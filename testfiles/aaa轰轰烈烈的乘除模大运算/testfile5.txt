// 20个参数
int heavy_calc(int a1, int a2, int a3, int a4, int a5, 
               int a6, int a7, int a8, int a9, int a10,
               int a11, int a12, int a13, int a14, int a15,
               int a16, int a17, int a18, int a19, int a20) {
    
    // 混合各种常数运算
    int s1 = a1 * 3 + a2 / 3;
    int s2 = a3 * 5 + a4 / 5;
    int s3 = a5 * 8 + a6 / 8; // 2的幂次
    int s4 = a7 * 10 + a8 / 10;
    int s5 = a9 * 31 + a10 / 31; // 2^k - 1
    int s6 = a11 * 33 + a12 / 33; // 2^k + 1
    int s7 = a13 * 123 + a14 / 123;
    int s8 = a15 * -7 + a16 / -7;
    int s9 = a17 % 4 + a18 % 10;
    int s10 = a19 * 0 + a20 * 1;

    // 长表达式防止寄存器过早释放
    int res = s1 - s2 + s3 - s4 + s5 - s6 + s7 - s8 + s9 - s10;
    
    // 再次使用参数，确保它们是活的
    if (a1 > 0) {
        res = res + (a20 / 2);
    }
    
    return res;
}

int main() {
    int i;
    int total = 0;
    
    // 循环调用，传入变化的参数
    for (i = 0; i < 1000; i = i + 1) {
        int t = i % 100;
        // 传入一些容易触发边界的数：正数，负数，0
        int val = heavy_calc(
            t, t-50, t*2, t/2, -t,
            i, i+1, i+2, i+3, i+4,
            100, 200, 300, 400, 500,
            -10, -20, -30, -40, -50
        );
        total = (total + val) % 1000000;
    }
    printf("Total: %d\n", total);
    return 0;
}
