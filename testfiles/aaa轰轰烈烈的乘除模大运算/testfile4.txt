const int M = 128;
const int N = 128;
const int SIZE = 16384; // 128 * 128
int matrix[16384]; 

int main() {
    int i; 
    int j;
    int sum = 0;

    // 初始化
    for (i = 0; i < SIZE; i = i + 1) {
        matrix[i] = i % 10;
    }

    // 模拟卷积操作或图像处理，产生大量地址计算
    // index = i * 128 + j
    // 这里的 * 128 应该被优化为 << 7
    for (i = 1; i < M - 1; i = i + 1) {
        for (j = 1; j < N - 1; j = j + 1) {
            int center_idx = i * 128 + j;
            int up_idx = (i - 1) * 128 + j;
            int down_idx = (i + 1) * 128 + j;
            int left_idx = i * 128 + (j - 1);
            int right_idx = i * 128 + (j + 1);
            
            // 复杂的加权计算，混合乘除
            int val = matrix[center_idx] * 4;
            val = val + matrix[up_idx] / 2;   // / 2 -> shift
            val = val + matrix[down_idx] / 2;
            val = val + matrix[left_idx] / 2;
            val = val + matrix[right_idx] / 2;
            
            // 归一化
            val = val / 6; // / 6 -> magic number
            
            sum = (sum + val) % 10000;
        }
    }
    printf("Matrix Sum: %d\n", sum);
    return 0;
}
