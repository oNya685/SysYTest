int concat(int a0[],int b0[],int c0[])
{
    int i;
    i=0;
    for (; i<3; )
    {
        c0[i]=a0[i];
      
        i=i+1;
    }

    j=0;
    for (; j<3; )
    {
        c0[i]=b0[j];
        i=i+1;
        j=j+1;
    }

    return 0;

}
