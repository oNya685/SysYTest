int enc(int a)
{
   if(a>25)
   a=a+60;
   else
   {
       a=a-15;
   }
   
   return a;

}

int dec(int a)
{
    if (a>85)
    a=a-59;
    else
    {
        a=a+14;
    }

    return a;
    
}

int main()
{
    int a;
    a=400;
    int res;
    res=enc(a);
    res=dec(res);
    printf("%d", res);
    res = 10;
    printf("%d", res);
    return 0;
}
