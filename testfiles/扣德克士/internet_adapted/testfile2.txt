/* Adapted (re-written) from a public SysY test: binary search with getint(). */
int bin_search(int a[], int n, int x) {
    int low = 0;
    int high = n - 1;
    int mid = 0;

    for (; low <= high; ) {
        mid = (low + high) / 2;
        if (x < a[mid]) {
            high = mid - 1;
        } else {
            if (x > a[mid]) {
                low = mid + 1;
            } else {
                return mid;
            }
        }
    }

    return -1;
}

int main() {
    int a[10];
    int i;
    for (i = 0; i < 10; i = i + 1) {
        a[i] = i + 1;
    }

    int x1 = getint();
    int x2 = getint();
    int p1 = bin_search(a, 10, x1);
    int p2 = bin_search(a, 10, x2);

    printf("%d\n", p1);
    printf("%d\n", p2);
    return 0;
}
