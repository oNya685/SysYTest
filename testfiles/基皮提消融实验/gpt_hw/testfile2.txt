const int C = 7;
int A[3] = {1, 2, 3};

int sumN(int n) {
    int s = 0, i = 1;
    for (;i <= n;) { s = s + i; i = i + 1; }
    return s;
}

int main() {
    printf("22371220\n");         // 1)

    int idx = getint();
    for (;idx < 0;idx = idx + 3) {  }
    idx = idx % 3;                  // 规范化下标到 0..2

    int B[3]; 
    B[0] = 4; B[1] = 5; B[2] = 6;
    // int B[3] = {4, 5, 6};
    int dp = A[0]*B[0] + A[1]*B[1] + A[2]*B[2]; // 32
    // int dp = 32;
    int v = A[idx];

    int t = 0, i = 0;
    for (i = 0; i < 3; i = i + 1) { t = t + (A[i] + B[i]); } // 21
    // t = 21;
    printf("%d\n", A[0]);           // 2)
    printf("%d\n", B[2]);           // 3)
    printf("%d\n", dp);             // 4)
    printf("%d\n", v);              // 5)
    printf("%d\n", sumN(5));        // 6) 15
    printf("%d\n", t);              // 7) 21
    printf("%d\n", A[1]*B[1]);      // 8) 10
    printf("%d\n", C);              // 9) 7
    // printf("%d\n", 15);             // 6) 15
    // printf("%d\n", 21);             // 7) 21
    // printf("%d\n", 10);             // 8) 10
    // printf("%d\n", 7);              // 9) 7
    printf("%d", dp - t);           // 10) 11（不带 \n）
    // printf("%d", 11);               // 10) 11（不带 \n）
    return 0;
}
