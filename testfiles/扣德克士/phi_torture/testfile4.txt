/* phi_torture 4: if without else creates phi with previous value */
int main() {
    int i;
    int x = 5;
    int y = 0;

    for (i = 0; i < 6; i = i + 1) {
        if (i % 2 == 0) {
            x = x + i;
        }

        if (i % 3 == 1) {
            y = y + x;
        } else {
            y = y - x;
        }
    }

    printf("%d\n", x * 100 + y);
    return 0;
}
