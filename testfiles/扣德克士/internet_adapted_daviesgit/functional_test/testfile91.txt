int N;

int newline;

int factor(int n )
{
    int i;
    int sum;
    sum=0;
    i=1;
    for (; i<n+1; )
    {
        if(n%i == 0)
        {
            sum=sum+i;
        }
        i=i+1;
    }

    return sum;
}
