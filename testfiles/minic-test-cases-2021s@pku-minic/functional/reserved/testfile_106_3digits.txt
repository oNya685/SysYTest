{
    int i,j,k;
    i=1;
    for(; i<5;)
    {
    	j=1;
        for(; j<5;)
    	{
    		k=1;
    		for(; k<5;)

                if (i!=k && i!=j && j!=k) {
                    int tmp=100*i+10*j+k;

}
