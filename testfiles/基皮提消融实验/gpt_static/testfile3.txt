int tick() {
    static int x = 1;
    x = x + 1;
    printf("tick:%d\n", x);
    return x;
}

int call_chain(int depth) {
    if (depth == 0) {
        return 0;
    }
    tick();
    return call_chain(depth - 1);
}

int main() {
    int i;
    for (i = 0; i < 2; i = i + 1) {
        call_chain(3);
    }
    tick();
    return 0;
}
