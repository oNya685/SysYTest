int N;

int newline;

int split(int n ,int a[])
{
    int i;
    i=N-1;
    for (; i!=-1; )
    {
        a[i]=n%10;
        n=n/10;
        i=i-1;

    }

    return 0;
}

int main()
{
    N=4;
    newline=10;
    int i;
    int m;
    int b[4];
    m = 1478;
    m = split(m,b);
    int t;
    i=0;
    for (; i<4; )
    {
        t=b[i];
        printf("%d", t);
        printf("\n");
        i=i+1;

    }
    return 0;
}
